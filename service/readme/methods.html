Methods

init(options)

Description:
Initialize plugin with parameters. This method is analogue of more simple $('#id').voicebase(options)
Parameters:
Name Type Description
options JSON Object Object with plugin configuration. You can see list of parameters here
Example:

jQuery('#player').voicebase('init', {
    playerId: 'playerBlock',
    playerType: 'jwplayer',
    mediaID: 'mediaId',
    apikey: 'ABCDEFGH-1234-ABCD-1234-1A2B3C4D5F6G',
    password: 'you_vb_password'
});

ready()

Description:
Ready callback of plugin.
Returned value:
Example:

    jQuery('#player').voicebase('init', {
    playerId: 'playerBlock',
    playerType: 'jwplayer',
    mediaID: 'mediaId',
    apikey: 'ABCDEFGH-1234-ABCD-1234-1A2B3C4D5F6G',
    password: 'you_vb_password',
    ready: function(){
        console.log('Voicebase is ready!');
    }
});

getPosition()

Description:
Getting current time position
Returned value:
Type Description
Integer Position of media in ms
Example:

    var position = jQuery('#player').voicebase('getPosition');

getSearchString()

Description:
Getting value of search bar
Returned value:
Type Description
String search string
Example:

    var position = jQuery('#player').voicebase('getSearchString');

getSharePositionUrl()

Description:
Getting string with url with parameter vbt. Parameter 'vbt' is equal current position of media file in ms
Returned value:
Type Description
String Url for sharing
Example:

    var url = jQuery('#player').voicebase('getSharePositionUrl');
    // returned: "http://some_site.com/?vbt=1819"

getShareSearchStringUrl()

Description:
Getting string with url with parameter vbs. Parameter 'vbs' is equal current value of search string
Returned value:
Type Description
String Url for sharing
Example:

    var url = jQuery('#player').voicebase('getShareSearchStringUrl');
    // returned: "http://some_site.com/?vbs=keyword"

search(text)

Description:
Plugin will start searching text.
Parameters:
Name Type Description
text String Text for searching
Example:

    jQuery('#player').voicebase('search', 'some words');

position(time)

Description:
Player rewind to time.
Parameters:
Name Type Description
time integer Time in ms
Example:

    jQuery('#player').voicebase('position', 100);

destroy()

Description:
Destructor of plugin. It removes all html elements of plugin and all events.
Example:

    jQuery('#player').voicebase('destroy');

addCustomMarkers()

Description:
Adding custom user markers on timeline of plugin.
Parameters:
Name Type Description
markers Object {'time': 'name'} Object with markers
Example:

    jQuery('#player').voicebase('addCustomMarkers', {
        '5': 'marker1',
        '25': 'marker2',
        '50': 'marker3'
    });

showCustomMarkers()

Description:
Showing custom user markers.
Example:

    jQuery('#player').voicebase('showCustomMarkers');

hideCustomMarkers()

Description:
Hiding custom user markers.
Example:

    jQuery('#player').voicebase('hideCustomMarkers');

nextCustomMarkers()

Description:
Player rewind to position of next custom marker.
Example:

    jQuery('#player').voicebase('nextCustomMarkers');

prevCustomMarkers()

Description:
Player rewind to position of previous custom marker.
Example:

    jQuery('#player').voicebase('prevCustomMarkers');

addHighlightTranscript()

Description:
Add highlight snippets in transcription.
Parameters:
Name Type Description
snippets Array of Objects [{'startTime': Integer, 'endTime': Integer, 'color': 'hex'}] Array with coordinates of snippets
Example:

    jQuery('#player').voicebase('addHighlightTranscript', [{
        startTime: 0,
        endTime: 25,
        color: '#F7B1BC'
    }, {
        startTime: 50,
        endTime: 105,
        color: '#979AF5'
    }]);

showHighlightTranscript()

Description:
Showing highlight snippets.
Example:

    jQuery('#player').voicebase('showHighlightTranscript');

hideHighlightTranscript()

Description:
Hiding highlight snippets.
Example:

    jQuery('#player').voicebase('hideHighlightTranscript');

showAutoGeneratedKeywords()

Description:
Show auto generated keywords.
Example:

    jQuery('#player').voicebase('showAutoGeneratedKeywords');

hideAutoGeneratedKeywords()

Description:
Hide auto generated keywords.
Example:

    jQuery('#player').voicebase('hideAutoGeneratedKeywords');
